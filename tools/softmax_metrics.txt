# NCU metrics collected by kernels/softmax/scripts/profile.sh.
# Lines starting with "# section:" name the display section for following metrics.
# Verified with Nsight Compute 2025.2.0.0 (build 35613519), NVIDIA GeForce RTX 5080, driver 580.126.09.

# section: launch
# [作用] 判断 kernel 启动规模、每个 SM 分到的工作量，以及 occupancy 的静态限制因素。
gpu__time_duration.sum                                  # Kernel 持续时间；当前 ncu raw CSV 中单位通常为 us。
launch__grid_size                                       # Grid 总 block 数。
launch__block_size                                      # 每个 block 的线程数。
launch__thread_count                                    # 本次 launch 的总线程数。
launch__waves_per_multiprocessor                        # 每个 SM 的 block 波数；过低通常说明并行工作量不足或尾部利用率差。
launch__registers_per_thread                            # 每线程分配寄存器数；过高会压低可驻留 warp/block 数。
launch__shared_mem_per_block                            # 每 block shared memory 总量。
launch__occupancy_limit_blocks                          # occupancy 受每 SM 最大 block 数限制时的上限。
launch__occupancy_limit_registers                       # occupancy 受寄存器资源限制时的上限。
launch__occupancy_limit_shared_mem                      # occupancy 受 shared memory 限制时的上限。
launch__occupancy_limit_warps                           # occupancy 受每 SM 最大 warp 数限制时的上限。

# section: compute
# [作用] 判断 SM/SMSP 计算管线利用率。不能只靠一个百分比断定 compute-bound，要结合 memory 和 stall 一起看。
sm__throughput.avg.pct_of_peak_sustained_elapsed        # SM 整体吞吐占峰值比例，elapsed 周期口径。
sm__warps_active.avg.pct_of_peak_sustained_active       # 活跃 warp 占理论峰值比例，接近 achieved occupancy。
smsp__cycles_active.avg.pct_of_peak_sustained_elapsed   # SMSP active 周期占峰值比例，用于观察调度器侧忙碌程度。

# section: memory
# [作用] 判断 DRAM/L1/L2 数据量、带宽利用率和 global memory 合并访存情况。
dram__bytes.sum                                         # DRAM 总访问字节数，包含读写。
dram__bytes_op_read.sum                                 # DRAM 实际读取字节数。
dram__bytes_op_write.sum                                # DRAM 实际写入字节数。
dram__throughput.avg.pct_of_peak_sustained_elapsed      # DRAM 吞吐占峰值比例，elapsed 周期口径。
gpu__dram_throughput.avg.pct_of_peak_sustained_elapsed  # GPU 级 DRAM 吞吐占峰值比例。
gpu__compute_memory_throughput.avg.pct_of_peak_sustained_elapsed # Compute/Memory roofline 口径的 memory throughput，占峰值比例。
l1tex__t_bytes_pipe_lsu_mem_global_op_ld.sum            # LSU global load 经 L1TEX 的总字节数。
l1tex__t_bytes_pipe_lsu_mem_global_op_st.sum            # LSU global store 经 L1TEX 的总字节数。
l1tex__average_t_sectors_per_request_pipe_lsu_mem_global_op_ld.ratio # 每个 global load request 平均访问 sector 数；越接近理想合并访存越好。
l1tex__average_t_sectors_per_request_pipe_lsu_mem_global_op_st.ratio # 每个 global store request 平均访问 sector 数。
lts__t_bytes.sum                                        # L2 侧总传输字节数。

# section: shared_memory
# [作用] 判断 shared memory 请求量和 bank conflict。对不用 shared memory 的 kernel 这些值应接近 0。
l1tex__data_pipe_lsu_wavefronts_mem_shared_op_ld.sum    # Shared memory load wavefront 数。
l1tex__data_pipe_lsu_wavefronts_mem_shared_op_st.sum    # Shared memory store wavefront 数。
l1tex__data_bank_conflicts_pipe_lsu.sum                 # LSU 产生的 shared-memory bank conflict 总数。

# section: stall
# [作用] 识别 warp 发射停顿原因。数值是占 active warp 的比例，主要看相对大小和主导项。
smsp__warp_issue_stalled_barrier_per_warp_active.pct            # 等待 barrier/sync 导致的停顿。
smsp__warp_issue_stalled_long_scoreboard_per_warp_active.pct    # 等待较长延迟依赖，常见来源是 global/L2 访存返回。
smsp__warp_issue_stalled_short_scoreboard_per_warp_active.pct   # 等待较短延迟依赖，常见于 shared memory 或部分短延迟数据路径。
smsp__warp_issue_stalled_math_pipe_throttle_per_warp_active.pct # 数学管线吞吐/排队导致的发射受限。
smsp__warp_issue_stalled_mio_throttle_per_warp_active.pct       # MIO/LSU 相关资源节流，常和访存指令压力相关。
smsp__warp_issue_stalled_not_selected_per_warp_active.pct       # warp 已 ready 但本周期未被调度器选中。
smsp__warp_issue_stalled_no_instruction_per_warp_active.pct     # 无可发射指令，常和取指/控制流/依赖链有关。

# section: instruction
# [作用] 看指令总量和主要执行管线分布，用于比较不同 softmax 实现的指令开销。
sm__inst_executed.sum                                  # SM 层面执行指令总数。
sm__inst_executed_pipe_fma.sum                         # FMA/FP32 计算管线执行指令数。
sm__inst_executed_pipe_lsu.sum                         # Load/store 管线执行指令数。
sm__inst_executed_pipe_xu.sum                          # XU/SFU 等辅助执行管线指令数；可辅助观察 exp 等特殊操作压力。

# section: sass
# [作用] SASS 级浮点操作统计。用于比较加法、乘法、FMA 的实际线程指令数量。
sm__sass_thread_inst_executed_op_fadd_pred_on.sum      # predicate-on 的浮点加法线程指令数。
sm__sass_thread_inst_executed_op_fmul_pred_on.sum      # predicate-on 的浮点乘法线程指令数。
sm__sass_thread_inst_executed_op_ffma_pred_on.sum      # predicate-on 的 FMA 线程指令数。

# section: thread
# [作用] 分析线程级指令量、predicate 有效率和分支一致性。
smsp__thread_inst_executed.sum
smsp__thread_inst_executed_pred_on.sum
smsp__thread_inst_executed_per_inst_executed.ratio      # 每条 warp 指令平均对应的线程指令数，反映 predicate/活跃线程效率。
smsp__sass_average_branch_targets_threads_uniform.pct   # 分支目标一致性；越接近 100% 分支发散越少。
