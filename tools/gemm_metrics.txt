# NCU metrics collected by kernels/gemm/scripts/profile.sh.
# Lines starting with "# section:" name the display section for following metrics.
# Focus: CUDA-core SGEMM kernels from naive/shared/register/vectorized/double-buffer/ultimate.

# section: launch
# [作用] 判断 kernel 启动规模、每个 SM 的工作量，以及 occupancy 的静态限制因素。
gpu__time_duration.sum
launch__grid_size
launch__block_size
launch__thread_count
launch__waves_per_multiprocessor
launch__registers_per_thread
launch__shared_mem_per_block
launch__occupancy_limit_blocks
launch__occupancy_limit_registers
launch__occupancy_limit_shared_mem
launch__occupancy_limit_warps

# section: occupancy
# [作用] 判断实际驻留 warp 数，以及 register/shared memory 是否压低 occupancy。
sm__warps_active.avg.pct_of_peak_sustained_active

# section: compute
# [作用] 判断 CUDA core FP32/FMA 管线利用率。GEMM 的核心计算应主要体现在 FFMA/FMA 指标上。
sm__throughput.avg.pct_of_peak_sustained_elapsed
smsp__cycles_active.avg.pct_of_peak_sustained_elapsed
sm__inst_executed_pipe_fma.sum
sm__sass_thread_inst_executed_op_ffma_pred_on.sum
sm__sass_thread_inst_executed_op_fmul_pred_on.sum
sm__sass_thread_inst_executed_op_fadd_pred_on.sum

# section: memory_global
# [作用] 判断 global memory 读写量、DRAM/L2 利用率和合并访存质量。
dram__bytes.sum
dram__bytes_op_read.sum
dram__bytes_op_write.sum
dram__throughput.avg.pct_of_peak_sustained_elapsed
gpu__dram_throughput.avg.pct_of_peak_sustained_elapsed
gpu__compute_memory_throughput.avg.pct_of_peak_sustained_elapsed
l1tex__t_bytes_pipe_lsu_mem_global_op_ld.sum
l1tex__t_bytes_pipe_lsu_mem_global_op_st.sum
l1tex__average_t_sectors_per_request_pipe_lsu_mem_global_op_ld.ratio
l1tex__average_t_sectors_per_request_pipe_lsu_mem_global_op_st.ratio
lts__t_bytes.sum

# section: memory_shared
# [作用] 判断 shared memory load/store 压力，以及 register/vectorized/ultimate 中 bank conflict 情况。
l1tex__data_pipe_lsu_wavefronts_mem_shared_op_ld.sum
l1tex__data_pipe_lsu_wavefronts_mem_shared_op_st.sum
l1tex__data_bank_conflicts_pipe_lsu.sum

# section: instruction
# [作用] 看总指令量和 LSU/FMA 指令比例，辅助判断是访存指令压力还是计算压力主导。
sm__inst_executed.sum
sm__inst_executed_pipe_lsu.sum
sm__inst_executed_pipe_fma.sum
smsp__thread_inst_executed.sum
smsp__thread_inst_executed_pred_on.sum
smsp__thread_inst_executed_per_inst_executed.ratio
smsp__sass_average_branch_targets_threads_uniform.pct

# section: stall
# [作用] 识别 warp 发射停顿原因。GEMM 常关注 barrier、scoreboard、MIO throttle、math pipe throttle。
smsp__warp_issue_stalled_barrier_per_warp_active.pct
smsp__warp_issue_stalled_long_scoreboard_per_warp_active.pct
smsp__warp_issue_stalled_short_scoreboard_per_warp_active.pct
smsp__warp_issue_stalled_math_pipe_throttle_per_warp_active.pct
smsp__warp_issue_stalled_mio_throttle_per_warp_active.pct
smsp__warp_issue_stalled_not_selected_per_warp_active.pct
smsp__warp_issue_stalled_no_instruction_per_warp_active.pct
